I. create a function to plot weight chart

A. input ID= specific ID number to get a plot of one patient

B. input ID= 0 to get the plot of the patients in the same age group

a. select age range by inputing min_age and max_age
b. select gender by inputing “F” or “M”
#create function
#age "30" means less or equal to 30days age
#ID=0 for multiple patient plot
#Id=   , for specific patient plot(select age and gender) 
weightchartplot <- function(ID,min_age,max_age,gender=c("M", "F") ){
  if(ID==0){
    age_selection <- inputdata[ which(inputdata$AGE_IN_DAYS < max_age), ]
    age_selection <- inputdata[ which(inputdata$AGE_IN_DAYS < max_age & mydata$AGE_IN_DAYS > min_age) , ]
    
    if (gender == "F"){
      genderF <- age_selection[ which(age_selection$GENDER == "F"), ]
      genderselection<-genderF
    }
    
    if(gender == "M"){
      genderM <- age_selection[ which(age_selection$GENDER == "M"), ]
      genderselection<-genderM
    }
    
    ok1<-as.data.frame(table(genderselection$PATIENT_ID))
    colnames(ok1) <- c("PATIENT_ID","freq")
    threshhold <- 5  
    sub_ok1<-subset(ok1, ok1$freq >threshhold) 
    numbers<-nrow(sub_ok1)
    #plot All patient under age has >5 samples
    first<-genderselection[genderselection$PATIENT_ID %in% sub_ok1[1:nrow(sub_ok1),1],]
    plot(first$AGE_IN_DAYS,first$Kg,col=first$PATIENT_ID,main =paste("Age(days)", min_age,"to",max_age),  xlab = "Age in Days", ylab = "Weights(kg)")
    perc<-as.data.frame(quantile(first$Kg,na.rm = T, c(.05, .10, .20,0.3,0.4,0.5,0.6,0.7,0.8,0.9,1)) )
    return(perc)
boxplot(first$Kg)
    
  }else if(ID){
    individualdata<-subset(mydata,mydata$PATIENT_ID==ID)
    plot((individualdata$AGE_IN_DAYS)/30,individualdata$Kg,main =paste("ID=", ID),  xlab = "Age in Days", ylab = "Weights(kg)")
    
  }
}
